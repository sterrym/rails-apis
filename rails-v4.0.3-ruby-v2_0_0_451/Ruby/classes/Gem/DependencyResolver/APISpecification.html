def initialize(set, api_data)
  @set = set
  @name = api_data[:name]
  @version = Gem::Version.new api_data[:number]
  @dependencies = api_data[:dependencies].map do |name, ver|
    Gem::Dependency.new name, ver.split(/\s*,\s*/)
  end
end
