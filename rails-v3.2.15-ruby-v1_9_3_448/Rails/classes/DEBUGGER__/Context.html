def initialize
  if Thread.current == Thread.main
    @stop_next = 1
  else
    @stop_next = 0
  end
  @last_file = nil
  @file = nil
  @line = nil
  @no_step = nil
  @frames = []
  @finish_pos = 0
  @trace = false
  @catch = "StandardError"
  @suspend_next = false
end

def check_suspend
  while MUTEX.synchronize {
      if @suspend_next
        DEBUGGER__.waiting.push Thread.current
        @suspend_next = false
        true
      end
    }
  end
